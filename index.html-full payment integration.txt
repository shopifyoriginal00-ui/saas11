        // Payment Success Handler
        function paymentSuccess(transactionId) {
            // Close all modals
            document.querySelectorAll('[id$="Modal"]').forEach(modal => {
                modal.classList.add('hidden');
            });
            
            // Update success modal
            document.getElementById('transactionId').textContent = transactionId;
            document.getElementById('paidAmount').textContent = selectedPrice;
            
            // Save to localStorage
            localStorage.setItem('subscription', selectedPlan);
            localStorage.setItem('subscriptionDate', new Date().toISOString());
            localStorage.setItem('transactionId', transactionId);
            
            // Show success modal
            document.getElementById('successModal').classList.remove('hidden');
            
            // Send to backend
            saveSubscription(transactionId);
        }

        // Save Subscription to Backend
        async function saveSubscription(transactionId) {
            try {
                const response = await fetch('/api/save-subscription', {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json'
                    },
                    body: JSON.stringify({
                        plan: selectedPlan,
                        amount: selectedPrice,
                        transactionId: transactionId,
                        paymentMethod: paymentMethod,
                        timestamp: new Date().toISOString()
                    })
                });
                
                const data = await response.json();
                console.log('Subscription saved:', data);
            } catch (error) {
                console.error('Error saving subscription:', error);
            }
        }

        document.getElementById('cardNumber')?.addEventListener('input', function(e) {

            let formattedValue = value.match(/.{1,4}/g)?.join(' ') || value;
            e.target.value = formattedValue;
        });

        document.getElementById('expiry')?.addEventListener('input', function(e) {
            let value = e.target.value.replace(/\D/g, '');
            if (value.length >= 2) {
                value = value.slice(0, 2) + '/' + value.slice(2, 4);
            }
            e.target.value = value;
        });
